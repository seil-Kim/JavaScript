<script>
function onCheck(){
	if(!fm.name.value){
		alert("이름을 입력하세요");
		fm.name.focus();
		return false;
	}
	if(!fm.pw.value){
		alert('암호를 입력하세요');
		fm.pw.focus();
		return false;
	}
	if(!fm.tel2.value||!fm.tel3.value){
		alert('전화번호를 입력하세요');
		fm.tel2.focus();
		return false;
	}
	if(!fm.job.value){
		alert('직업을 선택하세요');
		return false;
	}
	if(!(fm.gender[0].checked||fm.gender[1].checked)){
		alert('성별을 선택하세요');
		return false;
	}
	var obj=fm.play;
	if(!(obj[0].checked||obj[1].checked||obj[2].checked||obj[3].checked)){
		alert('취미를 하나 이상 선택하세요');
		return false;
	}
	if(!fm.Intro.value){
		alert('자기소개를 입력하세요');
		fm.Intro.focus();
		return false;
	}
	return true;
}
</script>
