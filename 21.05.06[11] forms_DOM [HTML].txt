	<script>
		function computeScore(){
			var choise = new Array(); //학생 응답
			var answer = [4, 2]; //정답
			var sname = document.stuInfo.sName.value;
			var noti = document.getElementById('result');
			var score = 0;
			choise[0] = document.forms[1].p1.value;
			choise[1] = document.forms[2].p2.value;
			for(i=0; i<2; i++){
				if(parseInt(choise[i]) == parseInt(answer[i])) score = score + 10;
			}
			noti.innerHTML = sname + ":" + score + "점";
		}
	</script>
